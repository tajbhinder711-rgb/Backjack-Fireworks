function calculateHaul() {

// Update visible counts

document.getElementById('shell-val').innerText = shellQty;

// Math.js precise floating point evaluations
const cakeTotal = math.evaluate(${cakeQty} * 89.99); const shellTotal = math.evaluate(``${shellQty} * 119.99);
const fountainTotal = math.evaluate(${fountainQty} * 44.99); const rawTotal = math.evaluate(``${cakeTotal} + ${shellTotal} + ${fountainTotal});

// Weight calculation: Cakes ~8lbs, Shell Kits ~12lbs, Fountains ~5lbs
const estWeight = math.evaluate((${cakeQty} * 8) + (${shellQty} * 12) + (${fountainQty} * 5)`);

// VIP Points reward system (`$1 = 1 point)
const vipPoints = Math.floor(rawTotal);

// Update Dashboard outputs
document.getElementById('total-price').innerText = '$' + rawTotal.toFixed(2);
document.getElementById('total-weight').innerText = estWeight + ' lbs';
document.getElementById('vip-points').innerText = vipPoints + ' pts';
}
